63  void notify()
64  {
65  std::unique_lock<std::mutex> l(mutex_);
66  ++count_;
67  cv_.notify_one();
68  }
72  void wait()
73  {
74  std::unique_lock<std::mutex> l(mutex_);
75  cv_.wait(l, [this] { return count_ != 0; });
