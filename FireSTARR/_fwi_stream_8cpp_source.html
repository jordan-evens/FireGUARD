27 FwiStream::FwiStream(const vector<TIMESTAMP_STRUCT>& dates,
28  Startup* startup,
29  const double latitude,
30  vector<Weather> cur_member)
31 {
32  static const AccumulatedPrecipitation ZeroFfmcPrecipitationAmount(2);
33  static const AccumulatedPrecipitation ZeroDmcPrecipitationAmount(10);
34  assert(cur_member.size() <= dates.size());
35  int month = dates.at(0).month;
36  auto started = (nullptr != startup) ||
37  (month > 9 || (9 == month && 15 <= dates.at(0).day));
38  auto shutdown = !started;
39  static auto fake_startup = Startup("",
40  TIMESTAMP_STRUCT{},
41  topo::Point(0, 0),
42  0,
43  Ffmc(0),
44  Dmc(0),
45  Dc(0),
46  AccumulatedPrecipitation(0),
47  false);
48  // HACK: could be started but now ended and marked as started because of month/day
49  const Startup* const use_startup = (!started || nullptr == startup)
50  ? &fake_startup
51  : startup;
52  auto last_ffmc = use_startup->ffmc();
53  auto last_dmc = use_startup->dmc();
54  auto last_dc = use_startup->dc();
55  Dc shutdown_dc(-1);
56  AccumulatedPrecipitation since_shutdown_accumulation(-1);
57  for (size_t date_index = 0; date_index < cur_member.size(); ++date_index)
58  {
59  const auto& wx = cur_member.at(date_index);
60  const auto for_date = dates.at(date_index);
61  month = for_date.month;
62  if (date_index >= 2)
63  {
64  // if we rolled into next year then start checking for startup conditions again
65  if (shutdown && started && 1 == month)
66  {
67  started = false;
68  }
69  if (!started)
70  {
71  // try to see if we can start things
72  double sum = 0;
73  for (auto j = date_index - 2; j <= date_index; ++j)
74  {
75  sum += cur_member.at(j).tmp().asDouble();
76  }
77  const auto avg = sum / 3;
78  // start stations if average temperature is at or over 11C for 3 consecutive days
79  started = avg >= 11;
80  if (started)
81  {
82  // use default startup values for FWI indices
83  // FIX: check that we're doing this right - should these be yesterday's indices
84  // (i.e. lastValues) or today's?
85  last_ffmc = Ffmc(85.0);
86  last_dmc = Dmc(6.0);
87  // FIX: DC adjustment isn't correct over winter
88  if (Dc(0) <= shutdown_dc)
89  {
90  last_dc = Dc(15.0);
91  }
92  }
93  // we aren't shutdown if we just started
94  shutdown = !started;
95  if (started)
96  {
97  shutdown_dc = Dc(-1);
98  since_shutdown_accumulation = AccumulatedPrecipitation(-1);
99  }
100  }
101  else if (((!shutdown && month > 9)
102  || (9 == month && 15 <= for_date.day)) && date_index >= 2)
103  {
104  // try to see if we can stop things
105  double sum = 0;
106  for (auto j = date_index - 2; j <= date_index; ++j)
107  {
108  sum += cur_member.at(j).tmp().asDouble();
109  }
110  auto avg = sum / 3;
111  // shut down station if DMC < 10 and temperature <= 2.5C for 3 days
112  shutdown = Dmc(10) > last_dmc && 2.5 >= avg;
113  // check if past week has been 2.5C
114  // NOTE: need to be able to average at least 7 days
115  if (!shutdown && date_index >= 6)
116  {
117  // try to see if we can stop things
118  sum = 0;
119  for (auto j = date_index - 6; j <= date_index; ++j)
120  {
121  sum += cur_member.at(j).tmp().asDouble();
122  }
123  avg = sum / 7;
124  // shut down station if temperature <= 2.5C for 7 days
125  shutdown = 2.5 >= avg;
126  }
127  }
128  else if (10 <= month)
129  {
130  shutdown = true;
131  }
132  }
133  auto apcp = wx.apcp();
134  // have to add apcp because it isn't in accumulation yet
135  if (shutdown)
136  {
137  since_shutdown_accumulation += apcp;
138  }
139  // if no startup then we don't need to add since it's 0
140  if (0 == date_index && nullptr != startup)
141  {
142  // need to add in 0800 obs for day 1
143  apcp += startup->apcp0800();
144  }
145  Ffmc ffmc(wx.tmp(), wx.rh(), wx.wind().speed(), apcp, last_ffmc);
146  if (shutdown)
147  {
148  // HACK: if we got 2mm of pcp since shutdown then assume 0 indices
149  if (since_shutdown_accumulation > ZeroFfmcPrecipitationAmount)
150  {
151  ffmc = Ffmc(0.0);
152  }
153  else
154  {
155  // want to decrease the value but never increase it
156  ffmc = min(last_ffmc, ffmc);
157  }
158  }
159  ffmc = max(Ffmc(0.0), ffmc);
160  Dmc dmc(wx.tmp(), wx.rh(), apcp, last_dmc, month, latitude);
161  if (shutdown)
162  {
163  // HACK: if we got 10mm of pcp since shutdown then assume 0 indices
164  if (since_shutdown_accumulation > ZeroDmcPrecipitationAmount)
165  {
166  dmc = Dmc(0.0);
167  }
168  else
169  {
170  // want to decrease the value but never increase it
171  dmc = min(last_dmc, dmc);
172  }
173  }
174  dmc = max(Dmc(0.0), dmc);
175  Dc dc(0);
176  if (shutdown)
177  {
178  assert(Dc(0) <= shutdown_dc);
179  assert(AccumulatedPrecipitation(0) <= since_shutdown_accumulation);
180  const auto a = 1.0;
181  // NOTE: concerned about the lack of overwinter precipitation present
182  // in reanalysis data
183  // NOTE: use 0.9 due to deep duff fuels
184  const auto b = 0.9;
185  const auto smi_f = 800 * exp(-shutdown_dc.asDouble() / 400.0);
186  const auto smi_s = a * smi_f + b * since_shutdown_accumulation.asDouble();
187  dc = min(last_dc, Dc(400 * log(800 / smi_s)));
188  }
189  else
190  {
191  dc = Dc(wx.tmp(), apcp, last_dc, month, latitude);
192  }
193  dc = max(Dc(0.0), dc);
194  const Isi isi(wx.wind().speed(), ffmc);
195  const Bui bui(dmc, dc);
196  const Fwi fwi(isi, bui);
197  // need to add in check for station startup and shutdown
198  wx_.emplace_back(wx.tmp(),
199  wx.rh(),
200  wx.wind(),
201  apcp,
202  ffmc,
203  dmc,
204  dc,
205  isi,
206  bui,
207  fwi);
208  last_ffmc = ffmc;
209  last_dmc = dmc;
210  last_dc = dc;
211  if (started && !shutdown)
212  {
213  shutdown_dc = last_dc;
214  since_shutdown_accumulation = AccumulatedPrecipitation(0.0);
215  }
216  }
217 }
