340 template <typename T, typename V>
341 T binary_find(const T lower,
342  const T upper,
343  const double value,
344  const std::function<V(T)>& fct)
345 {
346  const auto mid = lower + (upper - lower) / 2;
347  if (lower == upper)
348  {
349  return lower;
350  }
351  if (fct(mid) < value)
352  {
353  return binary_find(lower, mid, value, fct);
354  }
355  return binary_find(mid + 1, upper, value, fct);
356 }
367 template <typename T, typename V>
368 T binary_find_checked(const T lower,
369  const T upper,
370  const double value,
371  const std::function<V(T)>& fct)
372 {
373  if (fct(lower) < value)
374  {
375  return lower;
376  }
377  if (fct(upper) >= value)
378  {
379  return upper;
380  }
381  return binary_find(lower, upper, value, fct);
382 }
