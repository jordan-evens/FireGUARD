62 static void make_threshold(vector<double>* thresholds,
63  mt19937* mt,
64  const Day start_day,
65  const Day last_date,
66  double (*convert)(double value))
67 {
68  const auto total_weight = Settings::thresholdScenarioWeight() +
69  Settings::thresholdDailyWeight() + Settings::thresholdHourlyWeight();
70  uniform_real_distribution<double> rand(0.0, 1.0);
71  const auto general = rand(*mt);
72  for (size_t i = start_day; i < MAX_DAYS; ++i)
73  {
74  const auto daily = rand(*mt);
75  for (auto h = 0; h < DAY_HOURS; ++h)
76  {
77  // generate no matter what so if we extend the time period the results
78  // for the first days don't change
79  const auto hourly = rand(*mt);
80  // only save if we're going to use it
81  if (i <= last_date)
82  {
83  // subtract from 1.0 because we want weight to make things more likely not less
84  // ensure we stay between 0 and 1
85  thresholds->at((i - start_day) * DAY_HOURS + h) =
86  convert(
87  max(0.0,
88  min(1.0,
89  1.0 -
90  (Settings::thresholdScenarioWeight() * general
91  + Settings::thresholdDailyWeight() * daily
92  + Settings::thresholdHourlyWeight() * hourly) / total_weight)));
93  }
94  }
95  }
96 }

131 Scenario* Scenario::reset(mt19937* mt_extinction,
132  mt19937* mt_spread,
133  util::SafeVector* final_sizes)
134 {
135  ++simulation_;
136  clear();
137  const auto num = (static_cast<size_t>(last_date_) - start_day_ + 1) * DAY_HOURS;
138  extinction_thresholds_.resize(num);
139  spread_thresholds_by_ros_.resize(num);
140  make_threshold(&extinction_thresholds_, mt_extinction, start_day_, last_date_);
141  make_threshold(&spread_thresholds_by_ros_,
142  mt_spread,

150  for (const auto& o : observers_)

156 void Scenario::evaluate(const Event& event)
157 {
158  logging::check_fatal(event.time() < current_time_,
159  "Expected time to be > %f but got %f",
160  current_time_,
161  event.time());
162  const auto& p = event.cell();
163  switch (event.type())
164  {
165  case Event::FIRE_SPREAD:
166  scheduleFireSpread(event);
167  break;
168  case Event::SAVE:
169  saveObservers(event.time());
170  saveStats(event.time());
171  break;
172  case Event::NEW_FIRE:
173  // HACK: don't do this in constructor because scenario creates this in its constructor
174  points_[p].emplace_back(easting(p), northing(p));
175  if (fuel::is_null_fuel(event.cell()))
176  {
177  logging::fatal("Trying to start a fire in non-fuel");
178  }
179  logging::verbose("Starting fire in fuel type %s at time %f",
180  fuel::FuelType::safeName(fuel::check_fuel(event.cell())),
181  event.time());
182  if (!survives(event.time(), event.cell(), event.timeAtLocation()))
183  {
184  const auto wx = weather(event.time());
185  logging::info("Didn't survive ignition in %s with weather %f, %f",
186  fuel::FuelType::safeName(fuel::check_fuel(event.cell())),
187  wx->ffmc(),
188  wx->dmc());
189  // HACK: we still want the fire to have existed, so set the intensity of the origin
190  }
191  // fires start with intensity of 1
192  burn(event, 1);
193  scheduleFireSpread(event);
194  break;
195  case Event::END_SIMULATION:
196  logging::verbose("End simulation event reached at %f", event.time());
197  endSimulation();
198  break;
199  default:
200  throw runtime_error("Invalid event type");
201  }
202 }

374 Scenario* Scenario::run(map<double, ProbabilityMap*>* probabilities)
375 {
376  CriticalSection _(Model::task_limiter);
377  probabilities_ = probabilities;
378  unburnable_ = POOL_BURNED_DATA.acquire();
379  logging::check_fatal(ran(), "Scenario has already run");
380  logging::debug("Running scenario %d.%04d", id_, simulation_);
381  // don't do this until we run so we don't allocate memory too soon
382  intensity_ = make_unique<IntensityMap>(model());
383  //surrounded_ = POOL_BURNED_DATA.acquire();
384  for (auto time : save_points_)
385  {
386  // NOTE: these happen in this order because of the way they sort based on type
387  addEvent(Event::makeSave(static_cast<double>(time)));
388  }
389  if (nullptr == perimeter_)
390  {
391  addEvent(Event::makeNewFire(start_time_, cell(*start_cell_)));
392  }
393  else
394  {
395  intensity_->applyPerimeter(*perimeter_);
396  const auto& env = model().environment();
397  for (const auto& location : perimeter_->edge())
398  {
399  const auto cell = env.cell(location.hash());
400  logging::check_fatal(fuel::is_null_fuel(cell), "Null fuel in perimeter");
401  points_[cell].emplace_back(easting(cell), northing(cell));
402  }
403  addEvent(Event::makeFireSpread(start_time_));
404  }
405  // HACK: make a copy of the event so it still exists after it gets processed
406  // NOTE: sorted so that EventSaveASCII is always just before this
407  // Only run until last time we asked for a save for
408  logging::verbose("Creating simulation end event for %f", last_save_);
409  addEvent(Event::makeEnd(last_save_));
410  // mark all original points as burned at start
411  for (auto& kv : points_)
412  {
413  const auto& location = kv.first;
414  // would be burned already if perimeter applied
415  if (canBurn(location))
416  {
417  const auto fake_event = Event::makeFireSpread(
418  start_time_,
419  static_cast<IntensitySize>(1),
420  location);
421  burn(fake_event, static_cast<IntensitySize>(1));
422  }
423  }
424  while (!scheduler_.empty())
425  {
426  evaluateNextEvent();
427  }
428  ++COMPLETED;
429  // HACK: use + to pull value out of atomic
430  logging::info("Completed scenario %d.%04d [%d of %d] with final size %0.1f ha (%zu)",
431  id_,
432  simulation_,
433  +COMPLETED,
434  +COUNT,
435  currentFireSize(),
436  weightedDsr());
437  ran_ = true;
438  unburnable_ = check_reset(unburnable_, POOL_BURNED_DATA);
439  return this;
440 }

449 inline void Scenario::checkCondense(vector<InnerPos>& a)
450 {
451  if (a.size() > Settings::maxCellPoints())
452  {
453  size_t n_pos = 0;
454  auto n = numeric_limits<double>::min();
455  size_t ne_pos = 0;
456  auto ne = numeric_limits<double>::min();
457  size_t e_pos = 0;
458  auto e = numeric_limits<double>::min();
459  size_t se_pos = 0;
460  auto se = numeric_limits<double>::min();
461  size_t s_pos = 0;
462  auto s = numeric_limits<double>::max();
463  size_t sw_pos = 0;
464  auto sw = numeric_limits<double>::min();
465  size_t w_pos = 0;
466  auto w = numeric_limits<double>::max();
467  size_t nw_pos = 0;
468  auto nw = numeric_limits<double>::min();
469  // should always be in the same cell so do this once
470  const auto cell_x = static_cast<Idx>(a[0].x);
471  const auto cell_y = static_cast<Idx>(a[0].y);
472  for (size_t i = 0; i < a.size(); ++i)
473  {
474  const auto& p = a[i];
475  const auto x = p.x - cell_x;
476  const auto y = p.y - cell_y;
477  if (y > n)
478  {
479  n_pos = i;
480  n = y;
481  }
482  else if (y < s)
483  {
484  s_pos = i;
485  s = y;
486  }
487  const auto cur_ne = x + y;
488  const auto cur_sw = (1 - x) + (1 - y);
489  if (cur_ne > ne)
490  {
491  ne_pos = i;
492  ne = cur_ne;
493  }
494  else if (cur_sw > sw)
495  {
496  sw_pos = i;
497  sw = cur_sw;
498  }
499  if (x > e)
500  {
501  e_pos = i;
502  e = x;
503  }
504  else if (x < w)
505  {
506  w_pos = i;
507  w = x;
508  }
509  const auto cur_se = x + (1 - y);
510  const auto cur_nw = (1 - x) + y;
511  if (cur_se > se)
512  {
513  se_pos = i;
514  se = cur_se;
515  }
516  else if (cur_nw > nw)
517  {
518  nw_pos = i;
519  nw = cur_nw;
520  }
521  }
524  //condensed.insert(a[n_pos]);
525  //condensed.insert(a[ne_pos]);
526  //condensed.insert(a[e_pos]);
527  //condensed.insert(a[se_pos]);
528  //condensed.insert(a[s_pos]);
529  //condensed.insert(a[sw_pos]);
530  //condensed.insert(a[w_pos]);
531  //condensed.insert(a[nw_pos]);
532  //a.clear();
533  //a.insert(a.begin(), condensed.begin(), condensed.end());
534  //condensed.clear();
535  // keeping points is easier and near as fast as checking for duplicates
536  // but we don't need a member variable to do it
537  a = {
538  a[n_pos],
539  a[ne_pos],
540  a[e_pos],
541  a[se_pos],
542  a[s_pos],
543  a[sw_pos],
544  a[w_pos],
545  a[nw_pos]
546  };
547  }

549 void Scenario::scheduleFireSpread(const Event& event)
550 {
551  const auto time = event.time();
552  //note("time is %f", time);
553  current_time_ = time;
554  const auto wx = weather(time);
555  // logging::note("%d points", points_->size());
556  const auto this_time = util::time_index(time);
557  const auto next_time = static_cast<double>(this_time + 1) / DAY_HOURS;
558  // should be in minutes?
559  const auto max_duration = (next_time - time) * DAY_MINUTES;
560  //note("time is %f, next_time is %f, max_duration is %f",
561  // time,
562  // next_time,
563  // max_duration);
564  const auto max_time = time + max_duration / DAY_MINUTES;
565  if (wx->ffmc().asDouble() < minimumFfmcForSpread(time))
566  {
567  addEvent(Event::makeFireSpread(max_time));
568  logging::verbose("Waiting until %f because of FFMC", max_time);
569  return;
570  }
571  if (current_time_index_ != this_time)
572  {
573  current_time_index_ = this_time;
574  offsets_.clear();
575  max_intensity_.clear();
576  max_ros_ = 0.0;
577  }
578  auto any_spread = false;
579  for (const auto& kv : points_)
580  {
581  const auto& location = kv.first;
582  // any cell that has the same fuel, slope, and aspect has the same spread
583  const auto key = location.key();
584  const auto seek_spreading = offsets_.find(key);
585  if (seek_spreading == offsets_.end())
586  {
587  // have not calculated spread for this cell yet
588  const SpreadInfo origin(*this, time, location, nd(time), wx);
589  // will be empty if invalid
590  offsets_.emplace(key, origin.offsets());
591  if (!origin.isNotSpreading())
592  {
593  any_spread = true;
594  max_ros_ = max(max_ros_, origin.headRos());
595  max_intensity_[key] = max(max_intensity_[key], origin.maxIntensity());
596  }
597  }
598  else
599  {
600  // already did the lookup so use the result
601  any_spread |= !seek_spreading->second.empty();
602  }
603  }
604  if (!any_spread || max_ros_ < Settings::minimumRos())
605  {
606  logging::verbose("Waiting until %f", max_time);
607  addEvent(Event::makeFireSpread(max_time));
608  return;
609  }
610  //note("Max spread is %f, max_ros is %f",
611  // Settings::maximumSpreadDistance() * cellSize(),
612  // max_ros_);
613  const auto duration = ((max_ros_ > 0)
614  ? min(max_duration,
615  Settings::maximumSpreadDistance() * cellSize() /
616  max_ros_)
617  : max_duration);
618  //note("Spreading for %f minutes", duration);
619  const auto new_time = time + duration / DAY_MINUTES;
620  for (auto& kv : points_)
621  {
622  const auto& location = kv.first;
623  const auto key = location.key();
624  auto& offsets = offsets_.at(key);
625  if (!offsets.empty())
626  {
627  for (auto& o : offsets)
628  {
629  // offsets in meters
630  const auto offset_x = o.x * duration;
631  const auto offset_y = o.y * duration;
632  //note("%f, %f", offset_x, offset_y);
633  for (auto& p : kv.second)
634  {
635  const InnerPos pos(p.x + offset_x, p.y + offset_y);
636  const auto for_cell = cell(pos);
637  if (!(fuel::is_null_fuel(for_cell) || (*unburnable_)[for_cell.hash()]))
638  {
639  point_map_[for_cell].emplace_back(pos);
640  }
641  }
642  }
643  }
644  else
645  {
646  // can't just keep existing points by swapping because something may have spread into this cell
647  auto& pts = point_map_[location];
648  pts.insert(pts.end(), kv.second.begin(), kv.second.end());
649  }
650  kv.second.clear();
651  }
652  vector<topo::Cell> erase_what{};
653  for (auto& kv : point_map_)
654  {
655  if (!kv.second.empty())
656  {
657  auto& for_cell = kv.first;
658  if (canBurn(for_cell) && max_intensity_[for_cell.key()] > 0)
659  {
660  // HACK: make sure it can't round down to 0
661  const auto intensity = static_cast<IntensitySize>(max(
662  1.0,
663  max_intensity_[for_cell.key()]));
664  const auto fake_event = Event::makeFireSpread(
665  new_time,
666  intensity,
667  for_cell);
668  burn(fake_event, intensity);
669  }
670  // check if this cell is surrounded by burned cells or non-fuels
671  // if surrounded then just drop all the points inside this cell
672  if (!(*unburnable_)[for_cell.hash()])
673  {
674  if (!isSurrounded(for_cell) &&
675  survives(new_time, for_cell, new_time - arrival_[for_cell]))
676  {
677  checkCondense(kv.second);
678  //checkHull(for_cell, kv.second);
679  std::swap(points_[for_cell], kv.second);
680  }
681  else
682  {
683  // whether it went out or is surrounded just mark it as unburnable
684  (*unburnable_)[for_cell.hash()] = true;
685  erase_what.emplace_back(for_cell);
686  }
687  }
688  kv.second.clear();
689  }
690  }
691  for (auto& c : erase_what)
692  {
693  point_map_.erase(c);
694  points_.erase(c);
695  }
696  logging::verbose("Spreading %d points until %f", points_.size(), new_time);
697  addEvent(Event::makeFireSpread(new_time));
698 }
