58 void read_directory(const wstring& name, vector<string>* v, const wstring& match)
59 {
60  auto pattern(name);
61  pattern.append(match);
62  WIN32_FIND_DATA data;
63  HANDLE h_find;
64  if ((h_find = FindFirstFile(pattern.c_str(), &data)) != INVALID_HANDLE_VALUE)
65  {
66  while (true)
67  {
68  v->push_back(widestring_to_string(name + wstring(data.cFileName)));
69  if (0 == FindNextFile(h_find, &data))
70  {
71  break;
72  }
73  }
74  FindClose(h_find);
75  }
76 }

115 void make_directory_recursive(const char* dir) noexcept
116 {
117  char tmp[256];
118  snprintf(tmp, sizeof tmp, "%s", dir);
119  const auto len = strlen(tmp);
120  if (tmp[len - 1] == '/')
121  tmp[len - 1] = 0;
122  for (auto p = tmp + 1; *p; ++p)
123  if (*p == '/')
124  {
125  *p = 0;
126  make_directory(tmp);
127  *p = '/';
128  }
129  make_directory(tmp);
130 }
