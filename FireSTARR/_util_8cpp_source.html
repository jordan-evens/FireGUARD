29 void read_directory(const string& name, vector<string>* v, const string& match)
30 {
31  string full_match = ".*/" + match;
32  logging::info(("Matching '" + full_match + "'").c_str());
33  static const std::regex re(full_match, std::regex_constants::icase);
34  for (const auto &entry : fs::directory_iterator(name))
35  {
36  logging::debug(("Checking if file: " + entry.path().string()).c_str());
37  if (fs::is_regular_file(entry))
38  {
39  logging::info(("Checking regex match: " + entry.path().string()).c_str());
40  if (std::regex_match(entry.path().string(), re))
41  {
42  v->push_back(entry.path());
43  }
44  }
45  }
46 }

85 void make_directory_recursive(const char* dir) noexcept
86 {
87  char tmp[256];
88  snprintf(tmp, sizeof tmp, "%s", dir);
89  const auto len = strlen(tmp);
90  if (tmp[len - 1] == '/')
91  tmp[len - 1] = 0;
92  for (auto p = tmp + 1; *p; ++p)
93  if (*p == '/')
94  {
95  *p = 0;
96  make_directory(tmp);
97  *p = '/';
98  }
99  make_directory(tmp);
100 }
