28 class BurnedMap final
29  : public data::GridMap<unsigned char>
30 {
31 public:
32  // implement this class so that we can filter by fuel cells but not expose the members
38  BurnedMap(const Grid<unsigned char, unsigned char>& perim_grid, const Environment& env)
39  : GridMap<unsigned char, unsigned char>(
40  *env.makeMap<unsigned char>(static_cast<unsigned char>(0)))
41  {
42  // HACK: fix offset if the perimeter raster is different than this one
43  logging::check_fatal(0 != strcmp(perim_grid.proj4().c_str(), this->proj4().c_str()),
44  "Invalid projection for input perimeter raster - %s instead of %s",
45  perim_grid.proj4().c_str(),
46  this->proj4().c_str());
47  logging::check_fatal(perim_grid.cellSize() != this->cellSize(),
48  "Invalid cell size for input perimeter raster - %f instead of %f",
49  perim_grid.cellSize(),
50  this->cellSize());
51  const auto offset_x = static_cast<Idx>((this->xllcorner() - perim_grid.xllcorner()) /
52  this->cellSize());
53  const auto perim_origin = static_cast<Idx>(perim_grid.rows() + perim_grid.yllcorner()
54  / this->cellSize());
55  const auto this_origin = static_cast<Idx>(this->rows() + this->yllcorner() / this->
56  cellSize());
57  const auto offset_y = static_cast<Idx>((perim_origin - this_origin));
58  // make sure we don't go out of bounds on grid
59  const auto min_column = static_cast<Idx>(offset_x < 0 ? abs(offset_x) : 0);
60  const auto max_columns = min(this->columns(), perim_grid.columns());
61  const auto min_row = static_cast<Idx>(offset_y < 0 ? abs(offset_y) : 0);
62  const auto max_rows = min(this->rows(),
63  static_cast<Idx>(perim_grid.rows() - abs(offset_y)));
64  logging::note("Correcting perimeter raster offset by %dx%d cells",
65  offset_x,
66  offset_y);
67  size_t count = 0;
68  // since it was read in as a vector we need to check all the cells
69  for (auto r = min_row; r < max_rows; ++r)
70  {
71  for (auto c = min_column; c < max_columns; ++c)
72  {
73  const auto loc = env.cell(r, c);
74  const Location fixed_loc(static_cast<Idx>(r + offset_y),
75  static_cast<Idx>(c + offset_x));
76  const auto value = perim_grid.at(fixed_loc);
77  if (value != perim_grid.noData() && !fuel::is_null_fuel(loc))
78  {
79  this->GridMap<unsigned char, unsigned char>::set(loc, value);
80  ++count;
81  }
82  }
83  }
84  for (auto& kv : data)
85  {
86  logging::check_fatal(fuel::is_null_fuel(env.cell(kv.first)),
87  "Null fuel in BurnedData");
88  }
89  logging::info("Loaded burned area of size %lu ha",
90  static_cast<size_t>(this->cellSize() / 100.0 * count));
91  }
92 };

105 Perimeter::Perimeter(const Location& location,
106  const size_t size,
107  const wx::FwiWeather&,
108  const Environment& env)
109 {
110  // NOTE: FwiWeather is unused but could change this to try doing length to breadth ratio
111  auto perim_grid = env.makeMap<unsigned char>(0);
112  // want to find cells in the area that fill up the size we're looking for
113  size_t count = 0;
114  // convert into number of cells
115  const auto num_cells = size / (100.0 * 100.0 / (perim_grid->cellSize()
116  * perim_grid->cellSize()));
117  auto max_distance = sqrt(num_cells / M_PI);
118  perim_grid->set(location, 1);
119  ++count;
120  // HACK: assume fuel for origin matches the rest of the fire
121  while (num_cells > count)
122  {
123  const auto range = static_cast<Idx>(ceil(max_distance));
124  for (auto x = -range; x <= range && num_cells > count; ++x)
125  {
126  for (auto y = -range; y <= range && num_cells > count; ++y)
127  {
128  // look at any cell that's within the range
129  if (sqrt(util::pow_int<2>(x) + util::pow_int<2>(y)) < max_distance)
130  {
131  const auto r = static_cast<Idx>(location.row() + x);
132  const auto c = static_cast<Idx>(location.column() + y);
133  const auto loc = env.cell(r, c);
134  if (1 != perim_grid->at(loc) && !fuel::is_null_fuel(loc))
135  {
136  perim_grid->set(loc, 1);
137  ++count;
138  }
139  }
140  }
141  }
142  max_distance += 0.1;
143  }
144  const BurnedMap burned(*perim_grid, env);
145  burned_ = burned.makeList();
146  edge_ = burned.makeEdge();
147 }
