145  [[nodiscard]] static ConstantGrid<T, V>* readTiff(const string& filename,
146  TIFF* tif,
147  GTIF* gtif,
148  const topo::Point& point,
149  std::function<T(int, int)> convert)
150  {
151  V min_value = std::numeric_limits<V>::max();
152  V max_value = std::numeric_limits<V>::min();
153  const GridBase grid_info = read_header<T>(tif, gtif);
154  const auto coordinates = grid_info.findCoordinates(point, false);
155  auto min_column = max(static_cast<Idx>(0),
156  static_cast<Idx>(std::get<1>(*coordinates) - MAX_COLUMNS /
157  static_cast<Idx>(2)));
158  if (min_column + MAX_COLUMNS >= grid_info.columns())
159  {
160  min_column = grid_info.columns() - MAX_COLUMNS;
161  }
162  const auto max_column = min_column + MAX_COLUMNS;
163  auto min_row = max(static_cast<Idx>(0),
164  static_cast<Idx>(std::get<0>(*coordinates) - MAX_COLUMNS /
165  static_cast<Idx>(2)));
166  if (min_row + MAX_COLUMNS >= grid_info.rows())
167  {
168  min_row = grid_info.rows() - MAX_COLUMNS;
169  }
170  const auto max_row = min_row + MAX_COLUMNS;
171  const Idx offset_x = -min_column;
172  const Idx offset_y = -min_row;
173  T no_data = convert(grid_info.nodata(), grid_info.nodata());
174  vector<T> values(MAX_COLUMNS * MAX_COLUMNS, no_data);
175  int tile_width;
176  int tile_length;
177  TIFFGetField(tif, TIFFTAG_TILEWIDTH, &tile_width);
178  TIFFGetField(tif, TIFFTAG_TILELENGTH, &tile_length);
179  const auto buf = _TIFFmalloc(TIFFTileSize(tif));
180  const tsample_t smp{};
181  for (auto h = 0; h < grid_info.rows(); h += tile_length)
182  {
183  const auto y_min = static_cast<Idx>(max(0, min_row - h));
184  const auto y_limit = min(tile_length,
185  min(static_cast<int>(grid_info.rows()), max_row) - h);
186  for (auto w = 0; w < grid_info.columns(); w += tile_width)
187  {
188  TIFFReadTile(tif, buf, static_cast<uint32>(w), static_cast<uint32>(h), 0, smp);
189  const auto x_min = static_cast<Idx>(max(0, min_column - w));
190  const auto x_limit = min(tile_width,
191  min(static_cast<int>(grid_info.columns()),
192  max_column) - w);
193  for (auto y = y_min; y < y_limit; ++y)
194  {
195  // read in so that (0, 0) has a hash of 0
196  const Idx i = MAX_COLUMNS - (static_cast<Idx>(h) + y + offset_y + 1);
197  for (auto x = x_min; x < x_limit; ++x)
198  {
199  const auto cur_hash = static_cast<HashSize>(i) * MAX_COLUMNS + w + x +
200  offset_x;
201  const auto offset = y * tile_length + x;
202  V cur = *(static_cast<V*>(buf) + offset);
203  min_value = min(cur, min_value);
204  max_value = max(cur, max_value);
205  values.at(cur_hash) = convert(cur, grid_info.nodata());
206  }
207  }
208  }
209  }
210  _TIFFfree(buf);
211  const auto new_xll = grid_info.xllcorner() - offset_x * grid_info.cellSize();
212  const auto new_yll = grid_info.yllcorner() + (static_cast<double>(grid_info.rows()) -
213  max_row) * grid_info.cellSize();
214  logging::check_fatal(new_yll < grid_info.yllcorner(),
215  "New yllcorner is outside original grid");
216  logging::note("Translated lower left is (%f, %f) from (%f, %f)",
217  new_xll,
218  new_yll,
219  grid_info.xllcorner(),
220  grid_info.yllcorner());
221  auto result = new ConstantGrid<T, V>(grid_info.cellSize(),
222  MAX_COLUMNS,
223  MAX_COLUMNS,
224  no_data,
225  grid_info.nodata(),
226  new_xll,
227  new_yll,
228  string(grid_info.proj4()),
229  std::move(values));
230  auto new_location = result->findCoordinates(point, true);
231  logging::check_fatal(nullptr == new_location, "Invalid location after reading");
232  logging::note("Coordinates are (%d, %d => %f, %f)",
233  std::get<0>(*new_location),
234  std::get<1>(*new_location),
235  std::get<0>(*new_location) + std::get<3>(*new_location) / 1000.0,
236  std::get<1>(*new_location) + std::get<2>(*new_location) / 1000.0);
237  logging::note("Values for %s range from %d to %d",
238  filename.c_str(),
239  min_value,
240  max_value);
241  return result;
242  }
