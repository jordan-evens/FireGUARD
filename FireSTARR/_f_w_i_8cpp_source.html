106  2.4,
107  0.4,
108  -1.6,
109  -1.6,
110  -1.6,
111  -1.6,
112  -1.6,
113  0.9,
114  3.8,
115  5.8
116  };
117  //' '/* Use Northern hemisphere numbers */
118  //' '/* something goes wrong with >= */
119  if (latitude > 15.0)
120  {
121  return LfN[month];

192 //******************************************************************************************
193 // Function Name: DayLength
194 // Description: Calculates latitude/date dependent day length for DMC calculation
195 // Parameters:
196 // Latitude is latitude in decimal degrees of calculation location
197 // Month is integer (1..12) value of month of year for calculation
198 //
199 //******************************************************************************************
200 static constexpr double day_length(const double latitude, const int month) noexcept
201 {
202  //'''/* Day Length Arrays for four different latitude ranges '*/
203  //''/*
204  //' Use four ranges which respectively span:
205  //' - 90N to 33 N
206  //' - 33 N to 0
207  //' - 0 to -30
208  //' - -30 to -90
210  if ((latitude <= 90) && (latitude > 33.0))
211  {
212  return DAY_LENGTH46_N.at(static_cast<size_t>(month) - 1);
213  }
214  if ((latitude <= 33.0) && (latitude > 15.0))
215  {
216  return DAY_LENGTH20_N.at(static_cast<size_t>(month) - 1);
217  }
218  if ((latitude <= 15.0) && (latitude > -15.0))
219  {
220  return 9.0;
221  }
222  if ((latitude <= -15.0) && (latitude > -30.0))
223  {
224  return DAY_LENGTH20_S.at(static_cast<size_t>(month) - 1);
225  }
226  if ((latitude <= -30.0) && (latitude >= -90.0))
227  {
228  return DAY_LENGTH40_S.at(static_cast<size_t>(month) - 1);
229  }
230  return logging::fatal<double>("Unable to calculate DayLength");
231 }
232 //******************************************************************************************
233 // Function Name: FFMC
234 // Description: Calculates today's Fine Fuel Moisture Code
235 // Parameters:
236 // temperature is the 12:00 LST temperature in degrees celsius
237 // rh is the 12:00 LST relative humidity in %
238 // wind is the 12:00 LST wind speed in kph
239 // rain is the 24-hour accumulated rainfall in mm, calculated at 12:00 LST
240 // ffmc_previous is the previous day's FFMC
241 //******************************************************************************************
242 static double calculate_ffmc(const Temperature& temperature,
243  const RelativeHumidity& rh,
244  const Speed& wind,
245  const AccumulatedPrecipitation& rain,
246  const Ffmc& ffmc_previous) noexcept
247 {
248  //'''/* 1 '*/
249  auto mo = 147.2 * (101.0 - ffmc_previous.asDouble()) /
250  (59.5 + ffmc_previous.asDouble());
251  if (rain.asDouble() > 0.5)
252  {
253  //'''/* 2 '*/
254  const auto rf = rain.asDouble() - 0.5;
255  //'''/* 3a '*/
256  auto mr = mo + 42.5 * rf * (exp(-100.0 / (251.0 - mo))) * (1 - exp(-6.93 / rf));
257  if (mo > 150.0)
258  {
259  //'''/* 3b '*/
260  mr += 0.0015 * util::pow_int<2>(mo - 150.0) * sqrt(rf);
261  }
262  if (mr > 250.0)
263  {
264  mr = 250.0;
265  }
266  mo = mr;
267  }
268  //'''/* 4 '*/
269  const auto find_m = [&temperature, &rh, &wind, &mo]() noexcept
270  {
271  const auto ed = 0.942 * pow(rh.asDouble(), 0.679) + 11.0 * exp(
272  (rh.asDouble() - 100.0) / 10.0) + 0.18 *
273  (21.1 - temperature.asDouble()) * (1.0 - exp(-0.115 * rh.asDouble()));
274  if (mo > ed)
275  {
276  //'''/* 6a '*/
277  const auto ko = 0.424 * (1.0 - pow(rh.asDouble() / 100.0, 1.7)) +
278  0.0694 * sqrt(wind.asDouble()) * (1.0 - util::pow_int<8>(rh.asDouble() / 100.0));
279  //'''/* 6b '*/
280  const auto kd = ko * 0.581 * exp(0.0365 * temperature.asDouble());
281  //'''/* 8 '*/
282  return ed + (mo - ed) * pow(10.0, -kd);
283  }
284  //'''/* 5 '*/
285  const auto ew = 0.618 * pow(rh.asDouble(), 0.753) + 10.0 * exp(
286  (rh.asDouble() - 100.0) / 10.0) +
287  0.18 * (21.1 - temperature.asDouble()) * (1.0 - exp(-0.115 * rh.asDouble()));
288  if (mo < ew)
289  {
290  //'''/* 7a '*/
291  const auto kl = 0.424 * (1.0 - pow((100.0 - rh.asDouble()) / 100.0, 1.7)) + 0.0694 *
292  sqrt(wind.asDouble()) * (1 - util::pow_int<8>((100.0 - rh.asDouble()) / 100.0));
293  //'''/* 7b '*/
294  const auto kw = kl * 0.581 * exp(0.0365 * temperature.asDouble());
295  //'''/* 9 '*/
296  return ew - (ew - mo) * pow(10.0, -kw);
297  }
298  return mo;
299  };
300  const auto m = find_m();
301  //'''/* 10 '*/
302  return (59.5 * (250.0 - m) / (147.2 + m));
303 }

312 //******************************************************************************************
313 // Function Name: DMC
314 // Description: Calculates today's Duff Moisture Code
315 // Parameters:
316 // temperature is the 12:00 LST temperature in degrees celsius
317 // rh is the 12:00 LST relative humidity in %
318 // rain is the 24-hour accumulated rainfall in mm, calculated at 12:00 LST
319 // dmc_previous is the previous day's DMC
320 // latitude is the latitude in decimal degrees of the location for calculation
321 // month is the month of Year (1..12) for the current day's calculations.
322 //******************************************************************************************
323 static double calculate_dmc(const Temperature& temperature,
324  const RelativeHumidity& rh,
325  const AccumulatedPrecipitation& rain,
326  const Dmc& dmc_previous,
327  const int month,
328  const double latitude) noexcept
329 {
330  auto previous = dmc_previous.asDouble();
331  if (rain.asDouble() > 1.5)
332  {
333  //'''/* 11 '*/
334  const auto re = 0.92 * rain.asDouble() - 1.27;
335  //'''/* 12 '*/
336  const auto mo = 20.0 + exp(5.6348 - previous / 43.43);
337  const auto b = (previous <= 33.0)
338  ? //'''/* 13a '*/
339  100.0 / (0.5 + 0.3 * previous)
340  : ((previous <= 65.0)
341  ? //'''/* 13b '*/
342  14.0 - 1.3 * (log(previous))
343  : //'''/* 13c '*/
344  6.2 * log(previous) - 17.2);
345  //'''/* 14 '*/
346  const auto mr = mo + 1000.0 * re / (48.77 + b * re);
347  //'''/* 15 '*/
348  const auto pr = 244.72 - 43.43 * log(mr - 20.0);
349  previous = max(pr, 0.0);
350  }
351  const auto k = (temperature.asDouble() > -1.1)
352  ? 1.894 * (temperature.asDouble() + 1.1) * (100.0 - rh.asDouble()) *
353  day_length(latitude, month) * 0.000001
354  : 0.0;
355  //'''/* 17 '*/
356  return (previous + 100.0 * k);
357 }

367 //******************************************************************************************
368 // Function Name: DC
369 // Description: Calculates today's Drought Code
370 // Parameters:
371 // temperature is the 12:00 LST temperature in degrees celsius
372 // rain is the 24-hour accumulated rainfall in mm, calculated at 12:00 LST
373 // dc_previous is the previous day's DC
374 // latitude is the latitude in decimal degrees of the location for calculation
375 // month is the month of Year (1..12) for the current day's calculations.
376 //******************************************************************************************
377 static double calculate_dc(const Temperature& temperature,
378  const AccumulatedPrecipitation& rain,
379  const Dc& dc_previous,
380  const int month,
381  const double latitude) noexcept
382 {
383  auto previous = dc_previous.asDouble();
384  if (rain.asDouble() > 2.8)
385  {
386  //'/* 18 */
387  const auto rd = 0.83 * (rain.asDouble()) - 1.27;
388  //'/* 19 */
389  const auto qo = 800.0 * exp(-previous / 400.0);
390  //'/* 20 */
391  const auto qr = qo + 3.937 * rd;
392  //'/* 21 */
393  const auto dr = 400.0 * log(800.0 / qr);
394  previous = (dr > 0.0) ? dr : 0.0;
395  }
396  const auto lf = day_length_factor(latitude, month - 1);
397  //'/* 22 */
398  const auto v = max(0.0,
399  (temperature.asDouble() > -2.8)
400  ? 0.36 * (temperature.asDouble() + 2.8) + lf
401  : lf);
402  //'/* 23 */
403  const auto d = previous + 0.5 * v;
404  // HACK: don't allow negative values
405  return max(0.0, d);
406 }

422 static double calculate_isi(const Speed& wind, const Ffmc& ffmc) noexcept
423 {
424  //'''/* 24 '*/
425  const auto f_wind = exp(0.05039 * wind.asDouble());
426  //'''/* 1 '*/
427  const auto m = 147.2 * (101 - ffmc.asDouble()) / (59.5 + ffmc.asDouble());
428  //'''/* 25 '*/
429  const auto f_f = 91.9 * exp(-0.1386 * m) * (1.0 + pow(m, 5.31) / 49300000.0);
430  //'''/* 26 '*/
431  return (0.208 * f_wind * f_f);

460 static double calculate_bui(const Dmc& dmc, const Dc& dc) noexcept
461 {
462  if (dmc.asDouble() <= 0.4 * dc.asDouble())
463  {
464  // HACK: this isn't normally part of it but it's division by 0 without this
465  if (0 == dc.asDouble())
466  {
467  return 0;
468  }
469  //'''/* 27a '*/
470  return max(0.0,
471  0.8 * dmc.asDouble() * dc.asDouble() /
472  (dmc.asDouble() + 0.4 * dc.asDouble()));
473  }
474  //'''/* 27b '*/
475  return max(0.0,
476  dmc.asDouble() - (1.0 - 0.8 * dc.asDouble()
477  / (dmc.asDouble() + 0.4 * dc.asDouble()))
478  * (0.92 + pow(0.0114 * dmc.asDouble(), 1.7)));
479 }

507 static double calculate_fwi(const Isi& isi, const Bui& bui) noexcept
508 {
509  const auto f_d = (bui.asDouble() <= 80.0)
510  ? //'''/* 28a '*/
511  0.626 * pow(bui.asDouble(), 0.809) + 2.0

515  const auto b = 0.1 * isi.asDouble() * f_d;
516  if (b > 1.0)
517  {
518  //'''/* 30a '*/
519  return exp(2.72 * pow(0.434 * log(b), 0.647));

521  //'''/* 30b '*/

563 FwiWeather read(istringstream* iss, string* str)
564 {
565  // APCP
566  util::getline(iss, str, ',');
567  logging::extensive("APCP is %s", str->c_str());
568  const AccumulatedPrecipitation apcp(stod(str));
569  // TMP
570  util::getline(iss, str, ',');
571  logging::extensive("TMP is %s", str->c_str());
572  const Temperature tmp(stod(str));
573  // RH
574  util::getline(iss, str, ',');
575  logging::extensive("RH is %s", str->c_str());
576  const RelativeHumidity rh(stod(str));
577  // WS
578  util::getline(iss, str, ',');
579  logging::extensive("WS is %s", str->c_str());
580  const Speed ws(stod(str));
581  // WD
582  util::getline(iss, str, ',');
583  logging::extensive("WD is %s", str->c_str());
584  const Direction wd(stod(str), false);
585  // FFMC
586  util::getline(iss, str, ',');
587  logging::extensive("FFMC is %s", str->c_str());
588  const Ffmc ffmc(stod(str));
589  // DMC
590  util::getline(iss, str, ',');
591  logging::extensive("DMC is %s", str->c_str());
592  const Dmc dmc(stod(str));
593  // DC
594  util::getline(iss, str, ',');
595  logging::extensive("DC is %s", str->c_str());
596  const Dc dc(stod(str));
597  // ISI
598  util::getline(iss, str, ',');
599  logging::extensive("ISI is %s", str->c_str());
600  const Isi isi(stod(str), ws, ffmc);
601  // BUI
602  util::getline(iss, str, ',');
603  logging::extensive("BUI is %s", str->c_str());
604  const Bui bui(stod(str), dmc, dc);
605  // FWI
606  util::getline(iss, str, ',');
607  logging::extensive("FWI is %s", str->c_str());
608  const Fwi fwi(stod(str), isi, bui);
609  const Wind wind(wd, ws);
610  return {tmp, rh, wind, apcp, ffmc, dmc, dc, isi, bui, fwi};
611 }

617 double ffmc_effect(const Ffmc& ffmc) noexcept
618 {
619  const auto v = ffmc.asDouble();
620  const auto mc = 147.2 * (101.0 - v) / (59.5 + v);
621  return 91.9 * exp(-0.1386 * mc) * (1 + pow(mc, 5.31) / 49300000.0);
622 }
