46 /* Ellipsoid model constants (actual values here are for WGS84) */
47 constexpr auto SM_A = 6378137.0;
48 constexpr auto SM_B = 6356752.314;
49 constexpr auto UTM_SCALE_FACTOR = 0.9996;
50 double arc_length_of_meridian(const double phi) noexcept
51 {
52  const auto n = (SM_A - SM_B) / (SM_A + SM_B);
53  const auto alpha = ((SM_A + SM_B) / 2.0) * (1.0 + (pow_int<2>(n) / 4.0) + (pow_int<4>(n)
54  / 64.0));
55  const auto beta = (-3.0 * n / 2.0) + (9.0 * pow_int<3>(n) / 16.0) + (-3.0 * pow_int<5
56  >(n) / 32.0);
57  const auto gamma = (15.0 * pow_int<2>(n) / 16.0) + (-15.0 * pow_int<4>(n) / 32.0);
58  const auto delta = (-35.0 * pow_int<3>(n) / 48.0) + (105.0 * pow_int<5>(n) / 256.0);
59  const auto epsilon = (315.0 * pow_int<4>(n) / 512.0);
60  /* Now calculate the sum of the series and return */
61  return alpha * (phi + (beta * sin(2.0 * phi))
62  + (gamma * sin(4.0 * phi))
63  + (delta * sin(6.0 * phi))
64  + (epsilon * sin(8.0 * phi)));
65 }
66 constexpr double utm_central_meridian(const double zone)
67 {
68  return to_radians(utm_central_meridian_deg(zone));
69 }

74 double footpoint_latitude(const double y) noexcept
75 {
76  /* Precalculate n (Eq. 10.18) */
77  const auto n = (SM_A - SM_B) / (SM_A + SM_B);
78  /* Precalculate alpha_ (Eq. 10.22) */
79  /* (Same as alpha in Eq. 10.17) */
80  const auto alpha = ((SM_A + SM_B) / 2.0) * (1 + (pow_int<2>(n) / 4) + (pow_int<4>(n) /
81  64));
82  /* Precalculate row_ (Eq. 10.23) */
83  const auto y_alpha = y / alpha;
84  /* Precalculate beta_ (Eq. 10.22) */
85  const auto beta = (3.0 * n / 2.0) + (-27.0 * pow_int<3>(n) / 32.0) + (269.0 * pow_int<5
86  >(n) / 512.0);
87  /* Precalculate gamma_ (Eq. 10.22) */
88  const auto gamma = (21.0 * pow_int<2>(n) / 16.0) + (-55.0 * pow_int<4>(n) / 32.0);
89  /* Precalculate delta_ (Eq. 10.22) */
90  const auto delta = (151.0 * pow_int<3>(n) / 96.0) + (-417.0 * pow_int<5>(n) / 128.0);
91  /* Precalculate epsilon_ (Eq. 10.22) */
92  const auto epsilon = (1097.0 * pow_int<4>(n) / 512.0);
93  /* Now calculate the sum of the series (Eq. 10.21) */
94  return y_alpha + (beta * sin(2.0 * y_alpha))
95  + (gamma * sin(4.0 * y_alpha))
96  + (delta * sin(6.0 * y_alpha))
97  + (epsilon * sin(8.0 * y_alpha));
98 }
99 void map_lat_lon_to_xy(const double phi,
100  const double lambda,
101  const double lambda0,
102  double* x,
103  double* y) noexcept
104 {
105  const auto ep2 = (pow_int<2>(SM_A) - pow_int<2>(SM_B)) / pow_int<2>(SM_B);
106  const auto c = cos(phi);
107  const auto nu2 = ep2 * pow_int<2>(c);
108  const auto n = pow_int<2>(SM_A) / (SM_B * sqrt(1 + nu2));
109  const auto t = tan(phi);
110  const auto t2 = t * t;
111  const auto l = lambda - lambda0;
112  /* Precalculate coefficients for l**n in the equations below
113  so a normal human being can read the expressions for easting
114  and northing
115  -- l**1 and l**2 have coefficients of 1.0 */
116  const auto l3_coefficient = 1.0 - t2 + nu2;
117  const auto l4_coefficient = 5.0 - t2 + 9 * nu2 + 4.0 * (nu2 * nu2);
118  const auto l5_coefficient = 5.0 - 18.0 * t2 + (t2 * t2) + 14.0 * nu2 - 58.0 * t2 * nu2;
119  const auto l6_coefficient = 61.0 - 58.0 * t2 + (t2 * t2) + 270.0 * nu2 - 330.0 * t2 *
120  nu2;
121  const auto l7_coefficient = 61.0 - 479.0 * t2 + 179.0 * (t2 * t2) - (t2 * t2 * t2);
122  const auto l8_coefficient = 1385.0 - 3111.0 * t2 + 543.0 * (t2 * t2) - (t2 * t2 * t2);
123  /* Calculate easting (x) */
124  *x = n * c * l
125  + (n / 6.0 * pow_int<3>(c) * l3_coefficient * pow_int<3>(l))
126  + (n / 120.0 * pow_int<5>(c) * l5_coefficient * pow_int<5>(l))
127  + (n / 5040.0 * pow_int<7>(c) * l7_coefficient * pow_int<7>(l));
128  /* Calculate northing (y) */
129  *y = arc_length_of_meridian(phi)
130  + (t / 2.0 * n * pow_int<2>(c) * pow_int<2>(l))
131  + (t / 24.0 * n * pow_int<4>(c) * l4_coefficient * pow_int<4>(l))
132  + (t / 720.0 * n * pow_int<6>(c) * l6_coefficient * pow_int<6>(l))
133  + (t / 40320.0 * n * pow_int<8>(c) * l8_coefficient * pow_int<8>(l));
134 }
135 // The local variables Nf, nuf2, tf, and tf2 serve the same purpose as
136 // N, nu2, t, and t2 in MapLatLonToXY, but they are computed with respect
137 // to the footpoint latitude phi_f.
138 //
139 // x1_fraction, x2_fraction, x2_polynomial, x3_polynomial, etc. are to enhance
140 // readability and to optimize computations.
141 void map_xy_to_lat_lon(const double x,
142  const double y,
143  const double lambda0,
144  double* phi,
145  double* lambda) noexcept
146 {
147  /* Get the value of phi_f, the footpoint latitude. */
148  const auto phi_f = footpoint_latitude(y);
149  const auto ep2 = (pow_int<2>(SM_A) - pow_int<2>(SM_B)) / pow_int<2>(SM_B);
150  const auto cf = cos(phi_f);
151  const auto nuf2 = ep2 * pow_int<2>(cf);
152  /* Precalculate Nf and initialize nf_power */
153  const auto nf = pow_int<2>(SM_A) / (SM_B * sqrt(1 + nuf2));
154  auto nf_power = nf;
155  const auto tf = tan(phi_f);
156  const auto tf2 = tf * tf;
157  const auto tf4 = tf2 * tf2;
158  /* Precalculate fractional coefficients for x**n in the equations
159  below to simplify the expressions for latitude and longitude. */
160  const auto x1_fraction = 1.0 / (nf_power * cf);
161  nf_power *= nf; /* now equals Nf**2) */
162  const auto x2_fraction = tf / (2.0 * nf_power);
163  nf_power *= nf; /* now equals Nf**3) */
164  const auto x3_fraction = 1.0 / (6.0 * nf_power * cf);
165  nf_power *= nf; /* now equals Nf**4) */
166  const auto x4_fraction = tf / (24.0 * nf_power);
167  nf_power *= nf; /* now equals Nf**5) */
168  const auto x5_fraction = 1.0 / (120.0 * nf_power * cf);
169  nf_power *= nf; /* now equals Nf**6) */
170  const auto x6_fraction = tf / (720.0 * nf_power);
171  nf_power *= nf; /* now equals Nf**7) */
172  const auto x7_fraction = 1.0 / (5040.0 * nf_power * cf);
173  nf_power *= nf; /* now equals Nf**8) */
174  const auto x8_fraction = tf / (40320.0 * nf_power);
175  /* Precalculate polynomial coefficients for x**n.
176  -- x**1 does not have a polynomial coefficient. */
177  const auto x2_polynomial = -1.0 - nuf2;
178  const auto x3_polynomial = -1.0 - 2 * tf2 - nuf2;
179  const auto x4_polynomial = 5.0 + 3.0 * tf2 + 6.0 * nuf2 - 6.0 * tf2 * nuf2 - 3.0 * (nuf2
180  * nuf2)
181  - 9.0 * tf2 * (nuf2 * nuf2);
182  const auto x5_polynomial = 5.0 + 28.0 * tf2 + 24.0 * tf4 + 6.0 * nuf2 + 8.0 * tf2 *
183  nuf2;
184  const auto x6_polynomial = -61.0 - 90.0 * tf2 - 45.0 * tf4 - 107.0 * nuf2 + 162.0 * tf2
185  * nuf2;
186  const auto x7_polynomial = -61.0 - 662.0 * tf2 - 1320.0 * tf4 - 720.0 * (tf4 * tf2);
187  const auto x8_polynomial = 1385.0 + 3633.0 * tf2 + 4095.0 * tf4 + 1575 * (tf4 * tf2);
188  /* Calculate latitude */
189  *phi = phi_f + x2_fraction * x2_polynomial * (x * x)
190  + x4_fraction * x4_polynomial * pow_int<4>(x)
191  + x6_fraction * x6_polynomial * pow_int<6>(x)
192  + x8_fraction * x8_polynomial * pow_int<8>(x);
193  /* Calculate longitude */
194  *lambda = lambda0 + x1_fraction * x
195  + x3_fraction * x3_polynomial * pow_int<3>(x)
196  + x5_fraction * x5_polynomial * pow_int<5>(x)
197  + x7_fraction * x7_polynomial * pow_int<7>(x);
198 }
199 int lat_lon_to_utm(const Point& point, double* x, double* y) noexcept
200 {
201  const auto zone = static_cast<int>((point.longitude() + 180.0) / 6) + 1;
202  map_lat_lon_to_xy(to_radians(point.latitude()),
203  to_radians(point.longitude()),
204  utm_central_meridian(zone),
205  x,
206  y);
207  /* Adjust easting and northing for UTM system. */
208  *x = (*x) * UTM_SCALE_FACTOR + 500000.0;
209  *y = (*y) * UTM_SCALE_FACTOR;
210  if (*y < 0.0)
211  *y += 10000000.0;
212  return zone;
213 }

227 void utm_to_lat_lon(double x,
228  double y,
229  const int zone,
230  const bool is_southern_hemisphere,
231  double* lat,
232  double* lon) noexcept
233 {
234  x -= 500000.0;
235  x /= UTM_SCALE_FACTOR;
236  /* If in southern hemisphere, adjust y accordingly. */
237  if (is_southern_hemisphere)
238  y -= 10000000.0;
239  y /= UTM_SCALE_FACTOR;

241  map_xy_to_lat_lon(x, y, central_meridian, lat, lon);
