27 string find_value(const string& key, const string& within)
28 {
29  const auto c = within.find(key);
30  if (c != string::npos)
31  {
32  const string str = &within.c_str()[c + string(key).length()];
33  return str.substr(0, str.find(' '));
34  }
35  return "";
36 }

62 double find_meridian(const string& proj4) noexcept
63 {
64  try
65  {
66  int zone{};
67  auto meridian = 0.0;
68  if (find_value("+zone=", proj4, &zone, &str_to_int))
69  {
70  meridian = topo::utm_central_meridian_deg(zone);
71  }
72  else if (!find_value("+lon_0=", proj4, &meridian, &str_to_double))
73  {
74  logging::fatal("Can't find meridian for grid");
75  }
76  return meridian;
77  }
78  catch (...)
79  {
80  return logging::fatal<double>("Unable to parse meridian in string '%s'",
81  proj4.c_str());
82  }
83 }

115 void GridBase::createPrj(const string& dir, const string& base_name) const
116 {
117  ofstream out;
118  out.open(dir + base_name + ".prj");
119  logging::debug(proj4_.c_str());
120  // HACK: use what we know is true for the grids that were generated and
121  // hope it's correct otherwise
122  const auto proj = find_value("+proj=", proj4_);
123  if (!proj.empty() && proj != "tmerc" && proj != "utm")
124  {
125  logging::fatal("Cannot create projection for non-transverse grid");
126  }
127  out << "Projection TRANSVERSE\n";
128  out << "Datum AI_CSRS\n";
129  out << "Spheroid GRS80\n";
130  out << "Units METERS\n";
131  out << "Zunits NO\n";
132  out << "Xshift 0.0\n";
133  out << "Yshift 0.0\n";
134  out << "Parameters \n";
135  out << "0.9996 /* scale factor at central meridian\n";
136  // HACK: assume we're always on an exact degree

138  out << " 0 0 0.0 /* latitude of origin\n";
139  out << "500000.0 /* false easting (meters)\n";
140  out << "0.0 /* false northing (meters)\n";
141  out.close();

144 unique_ptr<Coordinates> GridBase::findCoordinates(const topo::Point& point,
145  const bool flipped) const
146 {
147  auto x = 0.0;
148  auto y = 0.0;
149  lat_lon_to_utm(point, this->zone(), &x, &y);
150  logging::note("Coordinates (%f, %f) converted to (%0.1f, %f, %f)",
151  point.latitude(),
152  point.longitude(),
153  this->zone(),
154  x,
155  y);
156  logging::debug("Lower left is (%f, %f)", this->xllcorner_, this->yllcorner_);
157  // convert coordinates into cell position
158  // use INT64 so we can see if coordinates are out of bounds
159  const auto actual_x = (x - this->xllcorner_) / this->cell_size_;
160  // these are already flipped across the y axis on reading so it's the same as for x now
161  auto actual_y = (y - this->yllcorner_) / this->cell_size_;
162  if (!flipped)
163  {
164  actual_y = static_cast<double>(this->rows_) - actual_y;
165  }
166  const auto column = static_cast<int64_t>(actual_x);
167  const auto row = static_cast<int64_t>(round(actual_y - 0.5));
168  if (0 > column || column >= columns_ || 0 > row || row >= rows_)
169  {
170  logging::debug("Returning nullptr fromfindCoordinates() for (%f, %f) => (%d, %d)",
171  actual_x,
172  actual_y,
173  column,
174  row);
175  return nullptr;
176  }
177  const auto sub_x = static_cast<SubSize>((actual_x - column) * 1000);
178  const auto sub_y = static_cast<SubSize>((actual_y - row) * 1000);
179  return make_unique<Coordinates>(static_cast<Idx>(row),
180  static_cast<Idx>(column),
181  sub_x,
182  sub_y);
183 }
