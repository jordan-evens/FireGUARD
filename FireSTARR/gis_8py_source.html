64 def Project(src, outputShapefile, outSpatialRef):
65  """!
66  Project to given spatial reference
67  @param src File to project
68  @param outputShapefile Destination file to save projection to
69  @param outSpatialRef Spatial reference to project into
70  @return None
71  """
72  # https://pcjericks.github.io/py-gdalogr-cookbook/projection.html
73  # input SpatialReference
74  # get the input layer
75  inDataSet = ogr.GetDriverByName('ESRI Shapefile').Open(src)
76  inLayer = inDataSet.GetLayer()
77  # create the CoordinateTransformation
78  inSpatialRef = GetSpatialReference(src)
79  #~ print inSpatialRef.ExportToWkt()
80  #~ print outSpatialRef.ExportToWkt()
81  coordTrans = osr.CoordinateTransformation(inSpatialRef, outSpatialRef)
82  # create the output layer
83  Delete(outputShapefile)
84  outDataSet = ogr.GetDriverByName('ESRI Shapefile').CreateDataSource(outputShapefile)
85  outLayer = outDataSet.CreateLayer(inLayer.GetName(), outSpatialRef, geom_type=ogr.wkbMultiPolygon)
86  # add fields
87  inLayerDefn = inLayer.GetLayerDefn()
88  for i in range(0, inLayerDefn.GetFieldCount()):
89  fieldDefn = inLayerDefn.GetFieldDefn(i)
90  if 'shape' not in fieldDefn.GetName().lower():
91  outLayer.CreateField(fieldDefn)
92  # get the output layer's feature definition
93  outLayerDefn = outLayer.GetLayerDefn()
94  # loop through the input features
95  inFeature = inLayer.GetNextFeature()
96  while inFeature:
97  # get the input geometry
98  geom = inFeature.GetGeometryRef()
99  # reproject the geometry
100  geom.Transform(coordTrans)
101  # create a new feature
102  outFeature = ogr.Feature(outLayerDefn)
103  # set the geometry and attribute
104  outFeature.SetGeometry(geom)
105  for i in range(0, outLayerDefn.GetFieldCount()):
106  outFeature.SetField(outLayerDefn.GetFieldDefn(i).GetNameRef(), inFeature.GetField(i))
107  # add the feature to the shapefile
108  outLayer.CreateFeature(outFeature)
109  # dereference the features and get the next input feature
110  del outFeature
111  inFeature = inLayer.GetNextFeature()
112  # Save and close the shapefiles
113  del coordTrans
114  del inSpatialRef
115  del inLayer
116  del inDataSet
117  del inFeature
118  del outLayerDefn
119  del inLayerDefn
120  del outLayer
121  del outDataSet
122  # create projection file for output
123  prj = os.path.splitext(outputShapefile)[0] + ".prj"
124  with open (prj, 'w') as file:
125  file.write(outSpatialRef.ExportToWkt())

168  """!
169  Convert a shapefile into a raster with the given spatial reference
170  @param shp Shapefile to convert to raster
171  @param raster Raster file path to save result to
172  @param reference Reference raster to use for extents and alignment
173  @return None
174  """
175  # https://gis.stackexchange.com/questions/222394/how-to-convert-file-shp-to-tif-using-ogr-or-python-or-gdal/222395
176  gdalformat = 'GTiff'
177  datatype = gdal.GDT_Byte
178  burnVal = 1 # value for the output image pixels
179  # Get projection info from reference image
180  ref_raster = gdal.Open(reference, gdal.GA_ReadOnly)
181  gt = ref_raster.GetGeoTransform()
182  pixelSizeX = gt[1]
183  pixelSizeY =-gt[5]
184  if pixelSizeX != pixelSizeY:
185  print "Raster must have square pixels"
186  sys.exit(-2)
187  # Open Shapefile
188  shapefile = ogr.Open(shp)
189  shapefile_layer = shapefile.GetLayer()
190  # Rasterise
191  #~ print("Rasterising shapefile...")
192  output = gdal.GetDriverByName(gdalformat).Create(raster, ref_raster.RasterXSize, ref_raster.RasterYSize, 1, datatype, options=['TFW=YES', 'COMPRESS=LZW', 'TILED=YES'])
193  output.SetProjection(ref_raster.GetProjectionRef())
194  output.SetGeoTransform(ref_raster.GetGeoTransform())
195  # Write data to band 1
196  band = output.GetRasterBand(1)
197  band.SetNoDataValue(0)
198  gdal.RasterizeLayer(output, [1], shapefile_layer, burn_values=[burnVal])
199  # Close datasets
200  del band
201  del output
202  del ref_raster
203  del shapefile
204  # create projection file for output
205  prj = os.path.splitext(raster)[0] + ".prj"
206  with open (prj, 'w') as file:
207  file.write(GetSpatialReference(shp).ExportToWkt())

253 def find_raster_meridians(year):

255  Find the meridians of input rasters for the given year
256  @param year Year to find raster meridians for
257  @return Dictionary of meridians to raster with each meridian

265  if not os.path.isdir(raster_root):
266  raster_root = default_root
267  rasters = [os.path.join(raster_root, x) for x in os.listdir(raster_root) if x[-4:].lower() == '.tif' and -1 != x.find('fuel')]
268  result = {}
269  for r in rasters:
270  raster = gdal.Open(r)
271  prj = raster.GetProjection()
272  srs = osr.SpatialReference(wkt=prj)
273  result[srs.GetProjParm('CENTRAL_MERIDIAN')] = r
274  del srs
275  del prj
276  del raster
277  MERIDIANS = result
278  if 0 == len(result):
279  logging.error("Error: missing rasters in directory {}".format(raster_root))
280  return result
