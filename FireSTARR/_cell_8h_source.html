52  [[nodiscard]] static constexpr Topo hashCell(const SlopeSize slope,
53  const AspectSize aspect,
54  const FuelCodeSize& fuel) noexcept

57  return static_cast<Topo>(fuel) << FuelShift
58  | static_cast<Topo>(slope) << SlopeShift
59  | static_cast<Topo>(0 == slope ? 0 : aspect) << AspectShift;

135  }

141  {
142  return static_cast<Topo>(topo_data_) & CellMask;
143  }
144 protected:
145  /*
146  * Field Natural Range Used Range Bits Bit Range
147  * Row 0 - 2047 0 - 2047 11 0 - 2047
148  * Column 0 - 2047 0 - 2047 11 0 - 2047
149  * PADDING 10
150  * Fuel 0 - 56 0 - 56 6 0 - 63
151  * Aspect 0 - 359 0 - 359 9 0 - 511
152  * Slope 0 - infinity 0 - 127 7 0 - 127
153  * Extra 10
154  *
155  * Rows and Columns are restricted to 2048 since that's what gets clipped out of
156  * the GIS outputs.
157  *
158  * Fuel is tied to how many variations of percent conifer/dead fir we want to use, and
159  * if we want to allow M1/M2/M3/M4 on their own, or just use the ones that are
160  * automatically tied to the green-up.
161  *
162  * Aspect is calculated to be in degrees, so 0 - 359.
163  *
164  * Slope is truncated to 0 - 60 because that's the range that affects effective wind
165  * speed for slopes, but there's an issue with this when it tries to calculate the
166  * horizontal rate of spread since then the slope has been truncated and the distance
167  * calculated will be wrong.
168  */
172  static constexpr uint32 FuelShift = LocationBits + 10;
176  static constexpr uint32 FuelBits = 6;
180  static constexpr Topo FuelBitMask = util::bit_mask<FuelBits, Topo>();
181  static_assert(FuelBitMask == 0x3F);
182  static_assert(FuelBitMask >= NUMBER_OF_FUELS);
186  static constexpr Topo FuelMask = FuelBitMask << FuelShift;
190  static constexpr uint32 AspectShift = FuelBits + FuelShift;
194  static constexpr uint32 AspectBits = 9;
198  static constexpr Topo AspectBitMask = util::bit_mask<AspectBits, Topo>();
199  static_assert(AspectBitMask == 0x1FF);
203  static constexpr Topo AspectMask = AspectBitMask << AspectShift;
207  static constexpr uint32 SlopeShift = AspectBits + AspectShift;
211  static constexpr uint32 SlopeBits = 7;
215  static constexpr Topo SlopeBitMask = util::bit_mask<SlopeBits, Topo>();
216  static_assert(SlopeBitMask == 0x7F);
217  static_assert(SlopeBitMask >= MAX_SLOPE_FOR_FACTOR);
221  static constexpr Topo SlopeMask = SlopeBitMask << SlopeShift;
225  static constexpr Topo CellMask = HashMask | FuelMask | AspectMask | SlopeMask;
226  static_assert(CellMask == 0x3FFFFF003FFFFF);
227 };
