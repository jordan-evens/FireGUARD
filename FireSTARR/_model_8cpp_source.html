105 void Model::readWeather(const fuel::FuelLookup& fuel_lookup,
106  const string& filename,
107  const bool for_actuals,
108  const wx::FwiWeather& yesterday)
109 {
110  map<size_t, map<Day, wx::FwiWeather>> wx{};
111  auto min_date = numeric_limits<Day>::max();
112  ifstream in;
113  in.open(filename);
114  if (in.is_open())
115  {
116  string str;
117  logging::info("Reading scenarios from '%s'", filename.c_str());
118  // read header line
119  getline(in, str);
120  // get rid of spaces
121  str.erase(std::remove(str.begin(), str.end(), ' '), str.end());
122  constexpr auto expected_header =
123  "Scenario,Date,APCP,TMP,RH,WS,WD,FFMC,DMC,DC,ISI,BUI,FWI";
124  logging::check_fatal(str != expected_header,
125  "Input CSV must have columns in this order:\n%s",
126  expected_header);
127  while (getline(in, str))
128  {
129  istringstream iss(str);
130  if (getline(iss, str, ',') && !str.empty())
131  {
132  // HACK: ignore date and just worry about relative order??
133  // Scenario
134  logging::verbose("Scenario is %s", str.c_str());
135  auto cur = ACTUALS;
136  if (!(str == "actual"))
137  {
138  try
139  {
140  cur = static_cast<size_t>(-stoi(str));
141  }
142  catch (std::exception&)
143  {
144  // HACK: somehow stoi() is still getting empty strings
145  logging::fatal("Error reading weather file %s", filename.c_str());
146  }
147  }
148  if (for_actuals && cur != ACTUALS)
149  {
150  // HACK: rely on actuals always being first
151  break;
152  }
153  if (wx.find(cur) == wx.end())
154  {
155  logging::debug("Loading scenario %d...", cur);
156  wx.emplace(cur, map<Day, wx::FwiWeather>());
157  }
158  auto& s = wx.at(cur);
159  struct tm t{};
160  util::read_date(&iss, &str, &t);
161  const auto ticks = mktime(&t);
162  if (1 == cur)
163  {
164  logging::debug("Date '%s' is %ld and calculated jd is %d",
165  str.c_str(),
166  ticks,
167  t.tm_yday);
168  if (!s.empty() && t.tm_yday < min_date)
169  {
170  logging::fatal(
171  "Weather input file crosses year boundary or dates are not sequential");
172  }
173  }
174  min_date = min(min_date, static_cast<Day>(t.tm_yday));
175  logging::check_fatal(s.find(static_cast<Day>(t.tm_yday)) != s.end(),
176  "Day already exists");
177  s.emplace(static_cast<Day>(t.tm_yday), wx::FwiWeather(&iss, &str));
178  }
179  }
180  in.close();
181  }
182  if (!for_actuals)
183  {
184  // HACK: it's added for everything right now, but alone if just doing actuals
185  const auto seek = wx.find(ACTUALS);
186  if (seek != wx.end())
187  {
188  wx.erase(ACTUALS);
189  }
190  }
191  // HACK: add yesterday into everything
192  for (auto& kv : wx)
193  {
194  auto& s = kv.second;
195  logging::check_fatal(s.find(static_cast<Day>(min_date - 1)) != s.end(),
196  "Day already exists");
197  s.emplace(static_cast<Day>(min_date - 1), yesterday);
198  }
199  // loop through and try to find duplicates
200  for (const auto& kv : wx)
201  {
202  const auto k = kv.first;
203  const auto s = kv.second;
204  if (wx_.find(k) == wx_.end())
205  {
206  const auto w = make_shared<wx::FireWeather>(fuel_lookup.usedFuels(), s);
207  wx_.emplace(k, w);
208  }
209  }
210 }
211 void Model::findStarts(const Location location)
212 {
213  logging::error("Trying to start a fire in non-fuel");
214  Idx range = 1;
215  while (starts_.empty())
216  {
217  for (Idx x = -range; x <= range; ++x)
218  {
219  for (Idx y = -range; y <= range; ++y)
220  {
221  // make sure we only look at the outside of the box
222  if (1 == range || abs(x) == range || abs(y) == range)
223  {
224  const auto loc = env_->cell(location.hash() + y + (x * MAX_COLUMNS));
225  if (!fuel::is_null_fuel(loc))
226  {
227  starts_.push_back(make_shared<topo::Cell>(cell(loc)));
228  }
229  }
230  }
231  }
232  ++range;
233  }
234  logging::info("Using %d start locations:", starts_.size());
235  for (const auto& s : starts_)
236  {
237  logging::info("\t%d, %d", s->row(), s->column());
238  }
239 }
240 void Model::makeStarts(Coordinates coordinates,
241  const topo::Point& point,
242  const wx::FwiWeather& yesterday,
243  const string& perim,
244  const size_t size)
245 {
246  const Location location(std::get<0>(coordinates), std::get<1>(coordinates));
247  if (!perim.empty())
248  {
249  logging::note("Initializing from perimeter %s", perim.c_str());
250  perimeter_ = make_shared<topo::Perimeter>(perim, point, *env_);
251  }
252  else if (size > 0)
253  {
254  logging::note("Initializing from size %d ha", size);
255  perimeter_ = make_shared<topo::Perimeter>(
256  cell(location),
257  size,
258  yesterday,
259  *env_);
260  }
261  // figure out where the fire can exist
262  if (nullptr != perimeter_ && !perimeter_->burned().empty())
263  {
264  logging::check_fatal(size != 0 && !perim.empty(), "Can't specify size and perimeter");
265  // we have a perimeter to start from
266  // HACK: make sure this isn't empty
267  starts_.push_back(make_shared<topo::Cell>(cell(location)));
268  logging::note("Fire starting with size %0.1f ha",
269  perimeter_->burned().size() * env_->cellSize() / 100.0);
270  }
271  else
272  {
273  if (nullptr != perimeter_)
274  {
275  logging::check_fatal(!perimeter_->burned().empty(),
276  "Not using perimeter so it should be empty");
277  logging::note("Using fire perimeter results in empty fire - changing to use point");
278  perimeter_ = nullptr;
279  }
280  logging::note("Fire starting with size %0.1f ha", env_->cellSize() / 100.0);
281  if (0 == size && fuel::is_null_fuel(cell(location.hash())))
282  {
283  findStarts(location);
284  }
285  else
286  {
287  starts_.push_back(make_shared<topo::Cell>(cell(location)));
288  }
289  }
290  logging::note("Creating %ld streams x %ld location%s = %ld scenarios",
291  wx_.size(),
292  starts_.size(),
293  starts_.size() > 1 ? "s" : "",
294  wx_.size() * starts_.size());
295 }
296 Iteration Model::readScenarios(const topo::StartPoint& start_point,
297  const double start,
298  const bool save_intensity,
299  const Day start_day,
300  const Day last_date)
301 {
302  vector<Scenario*> result{};
303 #pragma warning (push)
304 #pragma warning (disable: 4820)
305  auto saves = Settings::outputDateOffsets();
306  const auto setup_scenario = [&result, save_intensity, &saves](Scenario* scenario)
307  {
308  if (save_intensity)
309  {
310  scenario->registerObserver(new IntensityObserver(*scenario, ""));
311  scenario->registerObserver(new ArrivalObserver(*scenario));
312  scenario->registerObserver(new SourceObserver(*scenario));
313  }
314  // FIX: this should be relative to the start date, not the weather start date
315  for (const auto& i : saves)
316  {
317  scenario->addSaveByOffset(i);
318  }
319  result.push_back(scenario);
320  };
321 #pragma warning (pop)
322  for (const auto& kv : wx_)
323  {
324  const auto id = kv.first;
325  const auto cur_wx = kv.second;
326  if (nullptr != perimeter_)
327  {
328  setup_scenario(new Scenario(this,
329  id,
330  cur_wx.get(),
331  start,
332  perimeter_,
333  start_point,
334  start_day,
335  last_date));
336  }
337  else
338  {
339  for (const auto& cur_start : starts_)
340  {
341  // should always have at least the day before the fire in the weather stream
342  setup_scenario(new Scenario(this,
343  id,
344  cur_wx.get(),
345  start,
346  cur_start,
347  start_point,
348  start_day,
349  last_date));
350  }
351  }
352  }
353  return Iteration(result);
354 }

426 bool add_statistics(const size_t i,
427  vector<double>* means,
428  vector<double>* pct,
429  const Model& model,
430  const util::SafeVector& v)
431 {
432  const auto sizes = v.getValues();
433  logging::check_fatal(sizes.empty(), "No sizes at end of simulation");
434  const util::Statistics s{sizes};
435  static_cast<void>(util::insert_sorted(pct, s.percentile(95)));
436  static_cast<void>(util::insert_sorted(means, s.mean()));
437  if (model.isOutOfTime())
438  {
439  logging::note(
440  "Stopping after %d iterations. Time limit of %d seconds has been reached.",
441  i,
442  Settings::maximumTimeSeconds());
443  return false;
444  }
445  return true;
446 }
447 size_t runs_required(const size_t i,
448  const vector<double>* means,
449  const vector<double>* pct,
450  const Model& model)
451 {
452  if (model.isOutOfTime())
453  {
454  logging::note(
455  "Stopping after %d iterations. Time limit of %d seconds has been reached.",
456  i,
457  Settings::maximumTimeSeconds());
458  return 0;
459  }
460  const auto for_means = util::Statistics{*means};
461  const auto for_pct = util::Statistics{*pct};
462  if (!(i < Settings::minimumSimulationRounds()
463  || !for_means.isConfident(Settings::confidenceLevel())
464  || !for_pct.isConfident(Settings::confidenceLevel())))
465  {
466  return 0;
467  }
468  const auto left = max(for_means.runsRequired(i, Settings::confidenceLevel()),
469  for_pct.runsRequired(i, Settings::confidenceLevel()));
470  return left;
471 }
472 map<double, ProbabilityMap*> Model::runIterations(const topo::StartPoint& start_point,
473  const double start,
474  const Day start_day,
475  const bool save_intensity,
476  const bool for_actuals)
477 {
478  auto last_date = start_day;
479  for (const auto i : Settings::outputDateOffsets())
480  {
481  last_date = max(static_cast<Day>(start_day + i), last_date);
482  }
483  const auto seed = static_cast<unsigned int>(abs(
484  start * start_point.latitude() / start_point.longitude()));
485  // use independent seeds so that if we remove one threshold it doesn't affect the other
486  mt19937 mt_spread(seed);
487  mt19937 mt_extinction(seed * 3);
488  vector<double> means{};
489  vector<double> pct{};
490  size_t i = 0;
491  auto iterations = readScenarios(start_point,
492  start,
493  save_intensity,
494  start_day,
495  last_date);
496  // put probability maps into map
497  const auto saves = iterations.savePoints();
498  const auto started = iterations.startTime();
499  auto probabilities = make_prob_map(*this,
500  saves,
501  for_actuals,
502  started,
503  0,
504  Settings::intensityMaxLow(),
505  Settings::intensityMaxModerate(),
506  numeric_limits<int>::max());
507  auto runs_left = Settings::minimumSimulationRounds();
508  const auto recheck_interval = Settings::simulationRecheckInterval();
509  if (Settings::runAsync())
510  {
511  vector<Iteration> all_iterations{};
512  all_iterations.push_back(std::move(iterations));
513  size_t runs = 0;
514  const auto max_concurrent = Settings::concurrentSimulationRounds();
515  while (runs_left > 0)
516  {
517  // run what's left, up to min rounds at a time
518  const auto cur_runs = runs_left;
519  const auto total_runs = i + cur_runs;
520  while (min(cur_runs, max_concurrent) > all_iterations.size())
521  {
522  all_iterations.push_back(readScenarios(start_point,
523  start,
524  save_intensity,
525  start_day,
526  last_date));
527  }
528  auto threads = vector<std::thread>{};
529  auto all_scenarios = vector<Scenario*>{};
530  // need at least this many results
531  threads.reserve(max_concurrent);
532  for (size_t k = 0; k < min(cur_runs, all_iterations.size()); ++k)
533  {
534  logging::debug("Running iteration [%d of %d]", runs + 1, total_runs);
535  all_iterations.at(k).reset(&mt_extinction, &mt_spread);
536  auto& s = all_iterations.at(k).getScenarios();
537  all_scenarios.insert(all_scenarios.end(), s.begin(), s.end());
538  ++runs;
539  }
540  // sort in run so that they still get the same extinction thresholds as when unsorted
541  std::sort(all_scenarios.begin(),
542  all_scenarios.end(),
543  [](Scenario* lhs, Scenario* rhs) noexcept
544  {
545  // sort so that scenarios with highest dsrs are at the front
546  return lhs->weightedDsr() > rhs->weightedDsr();
547  });
548  for (auto s : all_scenarios)
549  {
550  threads.emplace_back(&Scenario::run,
551  s,
552  &probabilities);
553  }
554  for (auto& t : threads)
555  {
556  t.join();
557  }
558  for (size_t k = 0; k < min(cur_runs, all_iterations.size()); ++k)
559  {
560  auto& iteration = all_iterations.at(k);
561  auto final_sizes = iteration.finalSizes();
562  ++i;
563  //note("Adding %d", i);
564  if (!add_statistics(i, &means, &pct, *this, final_sizes))
565  {
566  // ran out of time
567  return probabilities;
568  }
569  }
570  runs_left = runs_required(i, &means, &pct, *this);
571  logging::note("Need another %d iterations", runs_left);
572  if (runs_left > recheck_interval)
573  {
574  runs_left = recheck_interval;
575  logging::note("Capping at %d iterations before checking again", runs_left);
576  }
577  }
578  // everything should be done when this section ends
579  }
580  else
581  {
582  logging::note("Running in synchronous mode");
583  while (runs_left > 0)
584  {
585  const auto cur_runs = runs_left;
586  size_t k = 0;
587  while (k < cur_runs)
588  {
589  logging::note("Running iteration %d", i + 1);
590  iterations.reset(&mt_extinction, &mt_spread);
591  for (auto s : iterations.getScenarios())
592  {
593  s->run(&probabilities);
594  }
595  ++i;
596  if (!add_statistics(i, &means, &pct, *this, iterations.finalSizes()))
597  {
598  // ran out of time
599  return probabilities;
600  }
601  ++k;
602  }
603  runs_left = runs_required(i, &means, &pct, *this);
604  logging::note("Need another %d iterations", runs_left);
605  if (runs_left > recheck_interval)
606  {
607  runs_left = recheck_interval;
608  logging::note("Capping at %d iterations before checking again", runs_left);
609  }
610  }
611  }
612  return probabilities;
613 }
614 int Model::runScenarios(const char* const output_directory,
615  const char* const weather_input,
616  const char* const fuels_table,
617  const char* const raster_root,
618  const wx::FwiWeather& yesterday,
619  const topo::StartPoint& start_point,
620  const tm& start_time,
621  const bool save_intensity,
622  const bool for_actuals,
623  const string& perimeter,
624  const size_t size)
625 {
626  const fuel::FuelLookup lookup(fuels_table);
627  auto env = topo::Environment::loadEnvironment(lookup,
628  raster_root,
629  start_point,
630  perimeter,
631  start_time.tm_year);
632  const auto position = env.findCoordinates(start_point, true);
633  logging::check_fatal(
634  std::get<0>(*position) > MAX_COLUMNS || std::get<1>(*position) > MAX_COLUMNS,
635  "Location loaded outside of grid at position (%d, %d)",
636  std::get<0>(*position),
637  std::get<1>(*position));
638  logging::info("Position is (%d, %d)", std::get<0>(*position), std::get<1>(*position));
639  const Location location{std::get<0>(*position), std::get<1>(*position)};
640  Model model(start_point, output_directory, &env);
641  auto x = 0.0;
642  auto y = 0.0;
643  const auto zone = lat_lon_to_utm(start_point, &x, &y);
644  logging::note("UTM coordinates are: %d %d %d",
645  zone,
646  static_cast<int>(x),
647  static_cast<int>(y));
648  logging::note("Grid has size (%d, %d)", env.rows(), env.columns());
649  logging::note("Fire start position is cell (%d, %d)",
650  location.row(),
651  location.column());
652  model.readWeather(lookup, weather_input, for_actuals, yesterday);
653  model.makeStarts(*position, start_point, yesterday, perimeter, size);
654  auto start_hour = ((start_time.tm_hour + (static_cast<double>(start_time.tm_min) / 60))
655  / DAY_HOURS);
656  // HACK: round to 2 digits so we can keep test output the same
657  start_hour = static_cast<double>(static_cast<int>(start_hour * 100)) / 100;
658  const auto start = start_time.tm_yday + start_hour;
659  const auto start_day = static_cast<Day>(start);
660  auto probabilities =
661  model.runIterations(start_point, start, start_day, save_intensity, for_actuals);
662  logging::note("Ran %d simulations", Scenario::completed());
663  show_probabilities(probabilities);
664  auto final_time = numeric_limits<double>::min();
665  for (const auto by_time : probabilities)
666  {
667  const auto time = by_time.first;
668  final_time = max(final_time, time);
669  const auto prob = by_time.second;
670  prob->saveAll(model, start_time, for_actuals, time, start_day);
671  }
672  for (const auto& kv : probabilities)
673  {
674  delete kv.second;
675  }
676  return 0;
677 }
678 }
