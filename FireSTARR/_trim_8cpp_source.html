26 void trim_left(string* s)
27 {
28  s->erase(s->begin(),
29  find_if(s->begin(),
30  s->end(),
31  [](const int ch) noexcept
32  {
33  return !isspace(ch);

36 // trim from end (in place)
37 void trim_right(string* s)
38 {
39  s->erase(find_if(s->rbegin(),
40  s->rend(),
41  [](const int ch) noexcept
42  {
43  return !isspace(ch);
44  }).base(),
45  s->end());
46 }
