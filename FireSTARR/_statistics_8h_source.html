216  explicit Statistics(vector<double> values)
217  {
218  // values should already be sorted
219  // std::sort(values.begin(), values.end());
220  n_ = values.size();
221  min_ = values[0];
222  max_ = values[n_ - 1];
223  median_ = values[n_ / 2];
224  const auto total_sum = std::accumulate(values.begin(),
225  values.end(),
226  0.0,
227  [](const double t, const double x)
228  {
229  return t + x;
230  });
231  mean_ = total_sum / n_;
232  for (size_t i = 0; i < percentiles_.size(); ++i)
233  {
234  const auto pos = std::min(n_ - 1,
235  static_cast<size_t>(truncl(
236  (static_cast<double>(i) / (percentiles_.size() - 1)) *
237  n_)));
238  // note("For %d values %dth percentile is at %d", n_, i, pos);
239  percentiles_[i] = values[pos];
240  }
241  const auto total = std::accumulate(values.begin(),
242  values.end(),
243  0.0,
244  [this](const double t, const double x)
245  {
246  return t + pow_int<2>(x - mean_);
247  });
248  standard_deviation_ = sqrt(total / n_);
249  sample_variance_ = total / (n_ - 1);
250  logging::check_fatal(min_ != percentiles_[0],
251  "Expected min to be %f not %f",
252  min_,
253  percentiles_[0]);
254  logging::check_fatal(max_ != percentiles_[100],
255  "Expected max to be %f not %f",
256  max_,
257  percentiles_[100]);
258  logging::check_fatal(median_ != percentiles_[50],
259  "Expected median to be %f not %f",
260  median_,
261  percentiles_[50]);
262  }
267  [[nodiscard]] double studentsT() const noexcept
268  {

273  return result;
274  }
280  [[nodiscard]] bool isConfident(const double relative_error) const noexcept
281  {
282  const auto st = studentsT();
283  const auto re = relative_error / (1 + relative_error);
284  // cout << st << " <= " << re << " is " << ((st <= re) ? "true" : "false") << endl;
285  return st <= re;

293  [[nodiscard]] size_t runsRequired(const size_t cur_runs,
294  const double relative_error) const
295  {
296  const auto re = relative_error / (1 + relative_error);
297  // HACK: set a limit just in case it wouldn't stop
298  const auto limit = sim::Settings::minimumSimulationRounds() * cur_runs;
299  const std::function<double(size_t)> fct = [this](const size_t i) noexcept
300  {
301  return T_VALUES[std::min(T_VALUES.size(), i) - 1]
302  * sqrt(sampleVariance() / i) / abs(mean());
303  };
304  return binary_find_checked(cur_runs, cur_runs + limit, re, fct) - cur_runs;
305  }
