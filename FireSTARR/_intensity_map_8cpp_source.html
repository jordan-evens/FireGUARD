105 bool IntensityMap::isSurrounded(const Location& location) const
106 {
107  // implement here so we can just lock once
108  lock_guard<mutex> lock(mutex_);
109  const auto x = location.column();
110  const auto y = location.row();
111  const auto min_row = static_cast<Idx>(max(y - 1, 0));
112  const auto max_row = min(y + 1, MAX_ROWS - 1);
113  const auto min_column = static_cast<Idx>(max(x - 1, 0));
114  const auto max_column = min(x + 1, MAX_COLUMNS - 1);
115  for (auto r = min_row; r <= max_row; ++r)
116  {
117  auto h = static_cast<size_t>(r) * MAX_COLUMNS + min_column;
118  for (auto c = min_column; c <= max_column; ++c)
119  {
120  // actually check x, y too since we care if the cell itself is burned
121  if (!(*is_burned_)[h])
122  {
123  return false;
124  }
125  ++h;
126  }
127  }
128  return true;
129 }
