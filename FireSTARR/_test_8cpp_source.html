163 int run_test(const char* output_directory,
164  const fuel::FuelLookup& fuel_lookup,
165  const string& fuel_name,
166  const SlopeSize slope,
167  const AspectSize aspect,
168  const int num_hours,
169  const wx::Dc& dc,
170  const wx::Bui& bui,
171  const wx::Dmc& dmc,
172  const wx::Ffmc& ffmc,
173  const wx::Wind& wind)
174 {
175  static const auto Latitude = 49.3911;
176  static const auto Longitude = -84.7395;
177  static const topo::StartPoint ForPoint(Latitude, Longitude);
178  const auto start_date = 123;
179  const auto end_date = start_date + static_cast<double>(num_hours) / DAY_HOURS;
180  util::make_directory_recursive(output_directory);
181  const auto fuel = fuel_lookup.byName(fuel_name);
182  auto values = vector<topo::Cell>();
183  values.reserve(MAX_ROWS * MAX_COLUMNS);
184  for (Idx r = 0; r < MAX_ROWS; ++r)
185  {
186  for (Idx c = 0; c < MAX_COLUMNS; ++c)
187  {
188  values.emplace_back(r, c, slope, aspect, fuel::FuelType::safeCode(fuel));
189  }
190  }
191  const auto cells = new data::ConstantGrid<topo::Cell>{
192  TEST_GRID_SIZE,
193  MAX_ROWS,
194  MAX_COLUMNS,
195  topo::Cell{},
196  -1,
197  TEST_XLLCORNER,
198  TEST_YLLCORNER,
199  TEST_PROJ4,
200  std::move(values)
201  };
202  TestEnvironment env(cells);
203  const Location start_location(static_cast<Idx>(MAX_ROWS / 2),
204  static_cast<Idx>(MAX_COLUMNS / 2));
205  Model model(ForPoint, output_directory, &env);
206  const auto start_cell = make_shared<topo::Cell>(model.cell(start_location));
207  TestWeather weather(fuel, start_date, dc, bui, dmc, ffmc, wind);
208  TestScenario scenario(&model, start_cell, ForPoint, start_date, end_date, &weather);
209  auto info = SpreadInfo(scenario,
210  start_date,
211  *start_cell,
212  model.nd(start_date),
213  weather.at(start_date));
214  map<double, ProbabilityMap*> probabilities{};
215  scenario.run(&probabilities);
216  scenario.saveObservers("");
217  logging::note("Final Size: %0.0f, ROS: %0.2f",
218  scenario.currentFireSize(),
219  info.headRos());
220  return 0;
221 }

229 int test(const int argc, const char* const argv[])
230 {
231  const wx::Dc dc(275);
232  const wx::Dmc dmc(35.5);
233  const wx::Bui bui(54, dmc, dc);
234  const wx::Ffmc ffmc(90);
235  assert(argc > 1 && 0 == strcmp(argv[1], "test"));
236  try
237  {
238  auto result = 0;
239  // HACK: use a variable and ++ so in case arg indices change
240  auto i = 1;
241  // start at 2 because first arg is "test"
242  ++i;
243  string output_directory(argv[i++]);
244  replace(output_directory.begin(), output_directory.end(), '\\', '/');
245  if ('/' != output_directory[output_directory.length() - 1])
246  {
247  output_directory += '/';
248  }
249  logging::debug("Output directory is %s", output_directory.c_str());
250  util::make_directory_recursive(output_directory.c_str());
251  const fuel::FuelLookup fuel_lookup(Settings::fuelLookupTable());
252  if (i == argc - 1 && 0 == strcmp(argv[i], "all"))
253  {
254  const auto num_hours = DEFAULT_HOURS;
255  constexpr auto mask = "%s%s_S%03d_A%03d_WD%03d_WS%03d/";
256  for (const auto& fuel : FUEL_NAMES)
257  {
258  auto simple_fuel_name{fuel};
259  simple_fuel_name.erase(
260  std::remove(simple_fuel_name.begin(), simple_fuel_name.end(), '-'),
261  simple_fuel_name.end());
262  simple_fuel_name.erase(
263  std::remove(simple_fuel_name.begin(), simple_fuel_name.end(), ' '),
264  simple_fuel_name.end());
265  simple_fuel_name.erase(
266  std::remove(simple_fuel_name.begin(), simple_fuel_name.end(), '('),
267  simple_fuel_name.end());
268  simple_fuel_name.erase(
269  std::remove(simple_fuel_name.begin(), simple_fuel_name.end(), ')'),
270  simple_fuel_name.end());
271  simple_fuel_name.erase(
272  std::remove(simple_fuel_name.begin(), simple_fuel_name.end(), '/'),
273  simple_fuel_name.end());
274  const auto out_length = output_directory.length() + 28 + simple_fuel_name.
275  length();
276  const auto out = new char[out_length];
277  for (SlopeSize slope = 0; slope <= 100; slope += SLOPE_INCREMENT)
278  {
279  for (AspectSize aspect = 0; aspect < 360; aspect += ASPECT_INCREMENT)
280  {
281  for (auto wind_direction = 0; wind_direction < 360; wind_direction +=
282  WD_INCREMENT)
283  {
284  const wx::Direction direction(wind_direction, false);
285  for (auto wind_speed = 0; wind_speed <= MAX_WIND; wind_speed += WS_INCREMENT
286  )
287  {
288  const wx::Wind wind(direction, wx::Speed(wind_speed));
289  logging::note(mask,
290  output_directory.c_str(),
291  simple_fuel_name.c_str(),
292  slope,
293  aspect,
294  wind_direction,
295  wind_speed);
296  sprintf_s(out,
297  out_length,
298  mask,
299  output_directory.c_str(),
300  simple_fuel_name.c_str(),
301  slope,
302  aspect,
303  wind_direction,
304  wind_speed);
305  result += run_test(out,
306  fuel_lookup,
307  fuel,
308  slope,
309  aspect,
310  num_hours,
311  dc,
312  bui,
313  dmc,
314  ffmc,
315  wind);
316  }
317  }
318  }
319  }
320  delete[] out;
321  }
322  }
323  else
324  {
325  const auto num_hours = argc > i ? stoi(argv[i++]) : DEFAULT_HOURS;
326  const auto slope = static_cast<SlopeSize>(argc > i ? stoi(argv[i++]) : 0);
327  const auto aspect = static_cast<AspectSize>(argc > i ? stoi(argv[i++]) : 0);
328  const wx::Speed wind_speed(argc > i ? stoi(argv[i++]) : 20);
329  // ReSharper disable CppAssignedValueIsNeverUsed

331  const wx::Wind wind(wind_direction, wind_speed);
332  // ReSharper restore CppAssignedValueIsNeverUsed
333  assert(i == argc);
334  logging::note("Running tests with constant inputs for %d:\n"
335  "\tSlope:\t\t\t%d\n"
336  "\tAspect:\t\t\t%d\n"
337  "\tWind Speed:\t\t%d\n"
338  "\tWind Direction:\t\t%d\n",
339  num_hours,
340  slope,
341  aspect,
342  wind_speed,
343  wind_direction);
344  result = run_test(output_directory.c_str(),
345  fuel_lookup,
346  "C-2",
347  slope,
348  aspect,
349  num_hours,
350  dc,
351  bui,
352  dmc,
353  ffmc,
